一个简单的 gatway 服务，基于nginx-proxy-manager
同时这个库提供了自动 ssl 续期


对应的 yaml
```
version: '3.8'

services:

app:

image: 'jc21/nginx-proxy-manager:latest'

restart: unless-stopped

ports:

- '8080:80'

- '8081:81'

- '8443:443'

volumes:

- ./data:/data

- ./letsencrypt:/etc/letsencrypt
```
